# Función para sacar un id pedido por un prompt
function getIdFromButton(): number{

    let input = prompt("Introduce el id")
    let id = Number(input)

    if(Number.isNaN(id)){
        alert("No has introducido un número")
        return 0
    }
    return id
}

# Función con .then / .catch
function findPais(id: number): void{  # Pasamos el id del pais que queremos buscar

    getPais(id) # Utilizamos las funcionse de antes para sacar el nombre del pais que conincida con el id que hemos dado
    .then((pais) => {
        console.log(pais, "pais") # Imprimimos en la consola el nombre
        salida.textContent = `El pais buscado es ${pais}` # Cambiamos el contenido del div con id salida al nombre del pais buscado
    })
    .catch((error) => {
        console.log(error, "error") # Imprimimos en la consola si hay error
        salida.textContent = error # Cambiamos el contenido del div con id salida al error
    })
}

# Función asíncrona 
async function findPaisesPorIdioma(idioma: string): Promise<void>{ # Pasamos el idioma de los paises que queremos buscar
    try{
        let paises = await getPaisesPorIdioma(idioma) # Utilizamos las funcionse de antes para sacar el nombre de los paises que conincida con el idioma que hemos dado
        console.log(paises, "paises") # Imprimimos en la consola los nombres

        salida.textContent = "Paises: " + paises.map(p => `${p.nombre}`).join(", ");  # Cambiamos el contenido del div con id salida a una cadena con los nombres de los paises

    }catch(error: any){
        console.log("Error:", error.message) # Imprimimos en la consola si hay error
        salida.textContent = "Idioma no encontrada" # Cambiamos el contenido del div con id salida al error
    }finally{
        console.log("Final") # Imprimimos en la consola que ha finalizado el proceso
    }

}
